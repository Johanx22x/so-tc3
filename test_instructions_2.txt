# Prueba con operaciones complejas, múltiples escrituras y lecturas

CREATE document.txt 2048
WRITE document.txt 0 "Inicio del documento con texto de prueba."
WRITE document.txt 100 "Añadiendo más contenido en una posición diferente."
WRITE document.txt 300 "Final del contenido del documento."

READ document.txt 0 40
READ document.txt 100 30
READ document.txt 300 35

CREATE notas.txt 1024
WRITE notas.txt 0 "Primera nota importante."
WRITE notas.txt 50 "Segunda nota en otra posición."
WRITE notas.txt 200 "Tercera nota al final de las notas."

READ notas.txt 0 20
READ notas.txt 50 25
READ notas.txt 200 30

LIST
DELETE document.txt
DELETE notas.txt
LIST
