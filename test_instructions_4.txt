# Prueba con combinaciones de operaciones y pruebas de espacio insuficiente

CREATE archivo_pequeno.txt 100
WRITE archivo_pequeno.txt 0 "Texto de prueba para archivo pequeño."
READ archivo_pequeno.txt 0 33

CREATE archivo_grande.txt 8192  # Esto debería fallar si el espacio es insuficiente
WRITE archivo_grande.txt 0 "Intento de escritura en un archivo que no debería crearse."

LIST

DELETE archivo_pequeno.txt

CREATE archivo_nuevo.txt 400
WRITE archivo_nuevo.txt 0 "Archivo creado después de liberar espacio."
